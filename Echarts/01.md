                axisTick: {
                    show: false // 去除刻度线
                },
                axisLabel: {
                    color: '#4c9bfd' // 文本颜色
                },
                splitLine: {
                    lineStyle: {
                        color: '#012f4a' // 分割线颜色
                    }
                }
                axisLine: {
                    show: false // 去除轴线
                },
                boundaryGap: false,// 去除轴内间距

                containLabel: true // 包含刻度文字在内

                textStyle: {
                    color: '#4c9bfd' // 图例文字颜色
                },

 series:
